JS字符串操作详解

概述

JavaScript中的字符串是不可变的，这意味着一旦创建了字符串，就不能更改其内容。对字符串的所有操作都会返回新的字符串，而不是修改原始字符串。理解这一点对于正确使用字符串方法非常重要。

在本教程中，我们将详细介绍JavaScript中字符串的各种操作方法，包括拼接、替换、截取、大小写转换、分割与连接、去除空格、字符访问和字符串填充等。

1. 字符串拼接 (Create)

1.1 使用 + 操作符

最基础的字符串拼接方式，使用加号操作符连接字符串。

let firstName = "John";
let lastName = "Doe";
let fullName = firstName + " " + lastName; // "John Doe"
console.log(fullName);

// 拼接数字和字符串
let age = 25;
let message = "我今年" + age + "岁"; // "我今年25岁"
console.log(message);
点击按钮查看结果

1.2 使用 += 操作符

可以将字符串追加到现有字符串变量的末尾。

let greeting = "Hello";
greeting += " World"; // 等同于 greeting = greeting + " World"
console.log(greeting); // "Hello World"

// 连续追加
let sentence = "JavaScript";
sentence += " is";
sentence += " awesome!";
console.log(sentence); // "JavaScript is awesome!"
点击按钮查看结果

1.3 使用模板字符串

使用反引号(`)和${}语法，可以更方便地嵌入变量和表达式。

let firstName = "John";
let lastName = "Doe";
let fullName = `Hello, ${firstName} ${lastName}!`; // "Hello, John Doe!"
console.log(fullName);

// 在模板字符串中使用表达式
let a = 5;
let b = 10;
let expression = `a + b = ${a + b}`; // "a + b = 15"
console.log(expression);

// 多行字符串
let multiline = `这是第一行
这是第二行
这是第三行`;
console.log(multiline);
点击按钮查看结果

1.4 使用concat()方法

concat()方法用于连接两个或多个字符串，返回新的字符串。

let str1 = "Hello";
let str2 = "World";
let result = str1.concat(" ", str2); // "Hello World"
console.log(result);

// 连接多个字符串
let part1 = "Java";
let part2 = "Script";
let part3 = " is";
let part4 = " fun!";
let fullString = part1.concat(part2, part3, part4); // "JavaScript is fun!"
console.log(fullString);

// 原始字符串不会改变
console.log(str1); // "Hello" (未改变)
点击按钮查看结果

2. 字符串替换 (Update)

2.1 使用replace()方法

replace()方法用于替换字符串中的第一个匹配项。

let text = "Hello World";
let newText = text.replace("World", "JavaScript"); // "Hello JavaScript"
console.log(newText);

// 替换数字
let price = "价格: $100";
let newPrice = price.replace("100", "80"); // "价格: $80"
console.log(newPrice);

// 原始字符串不会改变
console.log(text); // "Hello World" (未改变)
点击按钮查看结果

2.2 使用replaceAll()方法

replaceAll()方法用于替换字符串中所有匹配的项（ES2021新增）。

let phrase = "apple apple apple";
let newPhrase = phrase.replaceAll("apple", "orange"); // "orange orange orange"
console.log(newPhrase);

// 替换特殊字符
let sentence = "I love C++ and C++ is powerful";
let newSentence = sentence.replaceAll("C++", "JavaScript");
console.log(newSentence); // "I love JavaScript and JavaScript is powerful"

// 使用正则表达式
let text = "cat and Cat and cat";
let newText = text.replaceAll(/cat/gi, "dog"); // "dog and dog and dog"
console.log(newText);
点击按钮查看结果

3. 字符串截取 (Delete)

3.1 使用slice()方法

slice()方法提取字符串的一部分，返回新字符串，不会修改原字符串。

let str = "JavaScript";
let part1 = str.slice(0, 4); // "Java"
let part2 = str.slice(4); // "Script"
let part3 = str.slice(-6, -2); // "Scri"
console.log(part1, part2, part3);

// 截取邮箱用户名
let email = "user@example.com";
let username = email.slice(0, email.indexOf("@"));
console.log(username); // "user"

// 原始字符串不会改变
console.log(str); // "JavaScript" (未改变)
点击按钮查看结果

3.2 使用substring()方法

substring()方法类似于slice()，但不接受负参数。

let str = "JavaScript";
let part1 = str.substring(0, 4); // "Java"
let part2 = str.substring(4); // "Script"
let part3 = str.substring(2, 6); // "vaSc"
console.log(part1, part2, part3);

// 如果start大于end，会自动交换参数
let part4 = str.substring(6, 2); // "vaSc" (等同于substring(2, 6))
console.log(part4);
点击按钮查看结果

3.3 使用substr()方法

substr()方法已不推荐使用，但仍然可用。第一个参数是开始位置，第二个参数是要提取的字符数。

let str = "JavaScript";
let part1 = str.substr(0, 4); // "Java"
let part2 = str.substr(4); // "Script"
let part3 = str.substr(-6, 4); // "Scri"
console.log(part1, part2, part3);

// 注意：substr()方法已被标记为不推荐使用，建议使用slice()替代

5. 分割与连接 (Query/Create)

5.1 使用split()方法

split()方法将字符串分割成字符串数组。

// 按空格分割
let sentence = "Hello World JavaScript";
let words = sentence.split(" "); // ["Hello", "World", "JavaScript"]
console.log(words);

// 按逗号分割
let csv = "apple,banana,orange";
let fruits = csv.split(","); // ["apple", "banana", "orange"]
console.log(fruits);

// 限制分割数量
let text = "a-b-c-d-e";
let parts = text.split("-", 3); // ["a", "b", "c"]
console.log(parts);

// 分割每个字符
let word = "hello";
let chars = word.split(""); // ["h", "e", "l", "l", "o"]
console.log(chars);
点击按钮查看结果

5.2 使用join()方法

join()方法将数组元素连接成字符串。

// 使用空格连接
let words = ["Hello", "World", "JavaScript"];
let sentence = words.join(" "); // "Hello World JavaScript"
console.log(sentence);

// 使用逗号连接
let fruits = ["apple", "banana", "orange"];
let csv = fruits.join(","); // "apple,banana,orange"
console.log(csv);

// 使用其他分隔符
let numbers = [1, 2, 3, 4, 5];
let dashed = numbers.join("-"); // "1-2-3-4-5"
console.log(dashed);

// 不使用分隔符
let chars = ["h", "e", "l", "l", "o"];
let word = chars.join(""); // "hello"
console.log(word);
点击按钮查看结果

5.3 邮箱掩码示例

结合split()和join()实现邮箱掩码功能。

function maskEmail(email) {
let atIndex = email.indexOf("@");
let emailArr = email.split('');
for(let i = 1; i < atIndex - 1; i++) {
emailArr[i] = "*";
}
return emailArr.join('');
}

let masked = maskEmail("example@example.com"); // "e******@example.com"
console.log(masked);

// 更安全的邮箱掩码
function secureMaskEmail(email) {
let [name, domain] = email.split("@");
if (name.length <= 2) {
return "*@" + domain;
}
let maskedName = name[0] + "*".repeat(name.length - 2) + name[name.length - 1];
return maskedName + "@" + domain;
}

let secureMasked = secureMaskEmail("john@example.com"); // "j**n@example.com"
console.log(secureMasked);
点击按钮查看结果

6. 去除空格 (Delete)

6.1 trim()方法

去除字符串首尾的空白字符。

let textWithSpaces = " Hello World ";

// 处理制表符和换行符
let textWithWhitespace = "\t Hello \n World \n";
let trimmedWhitespace = textWithWhitespace.trim();
console.log(`"${trimmedWhitespace}"`); // "Hello \n World"

// 原始字符串不会改变
console.log(`"${textWithSpaces}"`); // " Hello World " (未改变)

7. 字符访问 (Query)

7.1 使用charAt()方法

返回指定位置的字符。

let word = "JavaScript";
let firstChar = word.charAt(0); // "J"
let lastChar = word.charAt(word.length - 1); // "t"
let middleChar = word.charAt(4); // "S"
console.log(firstChar, lastChar, middleChar);

// 访问不存在的索引
let outOfBounds = word.charAt(100); // "" (空字符串)
console.log(`"${outOfBounds}"`);
点击按钮查看结果

7.2 使用索引访问

使用方括号语法直接访问字符（ES5新增）。

let word = "JavaScript";
let firstChar = word[0]; // "J"
let lastChar = word[word.length - 1]; // "t"
let middleChar = word[4]; // "S"
console.log(firstChar, lastChar, middleChar);

// 访问不存在的索引
let outOfBounds = word[100]; // undefined
console.log(outOfBounds);

// 注意：不能通过索引修改字符串
word[0] = "P";
console.log(word); // "JavaScript" (未改变)
点击按钮查看结果

7.3 使用at()方法

at()方法支持负索引，可以更方便地访问末尾的字符（ES2022新增）。

let word = "JavaScript";
let lastChar = word.at(-1); // "t"
let secondLast = word.at(-2); // "p"
let firstChar = word.at(0); // "J"
console.log(lastChar, secondLast, firstChar);

// 与charAt()的对比
let charAtLast = word.charAt(word.length - 1); // "t"
let atLast = word.at(-1); // "t"
console.log(charAtLast, atLast); // 结果相同
点击按钮查看结果

8. 字符串填充 (Update)

8.1 padStart()方法

在字符串开头填充指定字符，直到达到指定长度。

let number = "5";
let paddedStart = number.padStart(3, "0"); // "005"
console.log(paddedStart);

// 用于格式化时间
let hour = "9";
let formattedHour = hour.padStart(2, "0"); // "09"
console.log(formattedHour);

// 使用其他字符填充
let text = "hello";
let paddedText = text.padStart(10, "*"); // "*****hello"
console.log(paddedText);

// 如果字符串长度已经足够或超过指定长度
let longText = "hello world";
let paddedLong = longText.padStart(5, "*"); // "hello world" (不会截断)
console.log(paddedLong);
点击按钮查看结果

8.2 padEnd()方法

在字符串末尾填充指定字符，直到达到指定长度。

let number = "5";

console.log(paddedEnd);

// 用于格式化表格
let name = "John";
let formattedName = name.padEnd(10, " "); // "John "
console.log(`"${formattedName}"`);

// 使用其他字符填充
let text = "hello";
let paddedText = text.padEnd(10, "*"); // "hello*****"

// 如果字符串长度已经足够或超过指定长度
let longText = "hello world";
let paddedLong = longText.padEnd(5, "*"); // "hello world" (不会截断)

注意事项

  • 字符串是不可变的，所有操作都返回新字符串，不会修改原字符串
  • 模板字符串使用反引号(`)而不是单引号或双引号
  • slice()支持负索引，而substring()不支持
  • substr()方法已被标记为不推荐使用，建议使用slice()替代
  • replaceAll()是ES2021新增方法，旧版本浏览器可能不支持
  • at()方法是ES2022新增方法，支持负索引访问
  • padStart()和padEnd()是ES2017新增方法
  • 使用索引访问字符串时，超出范围会返回undefined而不是空字符串
  • split()方法可以限制返回数组的长度
  • join()方法可以使用任意字符串作为分隔符，包括空字符串
